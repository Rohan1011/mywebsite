#include<stdio.h>
#include<omp.h>
Int main{
int n, thread;
printf(“Enter number of iterations:”);
scanf(“%d”,&n);
printf(“\n Enter the number of threads:”);
scanf(“%d”,&thread);
omp_set_num_threads(thread);
printf(“\n-----------------------------------------------\n”);
//Parallel for loop with static schedulingand chunk size 2
#pragma omp parallel for schedule(static, 4)
for(int i=0;i<n;i++){
int tid=omp_get_thread_num();
printf(“Thread %d : Iteration %d\n”, tid, i);
}
return 0;
}
