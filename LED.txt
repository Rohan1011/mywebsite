int ledPin[5] = {2, 3, 4, 5, 6}; // use actual digital pins

void setup() {
  for (int i = 0; i < 5; i++) {
    pinMode(ledPin[i], OUTPUT);
  }
}

void loop() {
  // Left to right
  for (int i = 0; i < 5; i++) {
    digitalWrite(ledPin[i], HIGH);
    delay(200);
    digitalWrite(ledPin[i], LOW);
  }

  // Right to left
  for (int i = 3; i > 0; i--) {   // skip first & last to avoid double blink
    digitalWrite(ledPin[i], HIGH);
    delay(200);
    digitalWrite(ledPin[i], LOW);
  }
